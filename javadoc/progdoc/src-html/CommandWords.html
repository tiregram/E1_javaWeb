001/**
002 * This class is part of the "World of Zuul" application. 
003 * "World of Zuul" is a very simple, text based adventure game.  
004 * 
005 * This class holds an enumeration table of all command words known to the game.
006 * It is used to recognise commands as they are typed in.
007 *
008 * @author  Michael Kolling and David J. Barnes + D.Bureau
009 * @version 2008.03.30 + 2014.02.09
010 */
011public class CommandWords
012{
013    // un tableau constant qui contient toutes les commandes valides :
014    private static final String[] sValidCommands = {
015        "go", "quit", "help","look","back","description","take","drop","invent"
016    };
017    // pour debuter, seulement 3 commandes numerotees 0, 1, et 2
018
019    /**
020     * Fonction booleenne pour verifier qu'une String donnee est une commande valide. 
021     * @return true si c'est le cas, et false sinon
022     */
023    public boolean isCommand( final String pString )
024    {
025        for ( int i=0; i<sValidCommands.length; i++ ) {
026            if ( sValidCommands[i].equals( pString ) )
027                return true;
028        } // for
029        // si on se retrouve la, c'est que pString n'a pas ete trouvee dns le tableau
030        return false;
031    } // isCommand()
032    
033    public String showAll(){
034        String vtt=" ";
035        for(String command : sValidCommands){
036            vtt+=(command+" ");
037        
038        }
039        return vtt;
040    }      
041    
042} // CommandWords
